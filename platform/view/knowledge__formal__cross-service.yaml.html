cross_constraints:
  - id: ske-volume-is-filestorage
    services: [container/ske, storage/filestorage]
    rule: "an SKE cluster's volume_id must be a storage/filestorage volume id (created first; torn down after the cluster)"
    provenance: VALIDATED
  - id: server-needs-networking-bundle
    services: [compute/virtualserver, networking/vpc, networking/security-group]
    rule: "a VM create needs live vpc + subnet (networking/vpc) and a security-group (networking/security-group) plus keypair/image/server-type lookups from its own service"
    provenance: VALIDATED
  - id: shared-path-roots-need-per-service-hosts
    services: [container/ske, database/mysql, database/mariadb, database/postgresql, database/epas, database/sqlserver, database/cachestore]
    rule: "path roots collide across services (/v1/clusters, /v1/parameters are used by ske + every DBaaS engine) — every call MUST target its own per-service host; coverage matching by (method, path) alone over-counts (each engine needs its own lifecycle)"
    provenance: VALIDATED
  - id: shared-subnet-fixed-ip-no-collision
    services: [networking/vpc, networking/loadbalancer]
    rule: "lifecycles adopting the ONE shared subnet must pin fixed host IPs at DISTINCT addresses (dependencies.json vpc_schedule.fixed_ip_map; .1 reserved for gateway) so parallel adopters never claim the same IP"
    provenance: VALIDATED
  - id: firewall-implicit-on-gateway-resources
    services: [networking/firewall, networking/vpc, networking/direct-connect, networking/loadbalancer]
    rule: "a Firewall is never created standalone — it is minted by enabling 'Firewall 사용' on an Internet Gateway, Transit Gateway (Uplink Firewall), Direct Connect or Load Balancer (userguide firewall how_to). Firewall-rule scenarios must first create one of those carriers and capture its firewall_id"
    provenance: docs
  - id: gslb-monitoring-needs-target-allowlist
    services: [networking/gslb, networking/firewall, networking/security-group]
    rule: "GSLB health monitoring only reaches a connection target if the target's Firewall AND Security Group allow the GSLB monitor traffic (userguide) — combo scenarios pairing GSLB with LB/VM targets need allow rules on both layers"
    provenance: docs
  - id: dc-prereq-security-group
    services: [networking/direct-connect, networking/security-group]
    rule: "Security Group is a documented 선행 서비스 of Direct Connect (userguide) — the DC create body itself only needs the VPC, but workloads behind the DC are expected to be SG-protected; keep an SG in DC combo scenarios"
    provenance: docs
